001package com.genband.kandy.api;
002
003/**
004 * Represent the class with Full Kandy and SPiDR Configuration like as:
005 * Kandy USer Info, Kandy API Configuration, Kandy SDK Configuration and SPiDR Configuration
006 * 
007 */
008public interface IKandyGlobalSettings {
009        
010        /**
011         * Set Kandy api secret<br>
012         * <b>Note:<br><i>Must be called each time when app starts</i></b>
013         * @param apiSecret to be set
014         */
015        public void setKandyDomainSecret(String apiSecret);
016
017        /**
018         * Set Kandy host url<br>
019         * <b>Note:<br><i>Must be called each time when app starts</i></b>
020         * @param url to be set
021         */
022        public void setKandyHostURL(String url);
023
024        /**
025         * Get Kandy Host URL 
026         * @return url of the Kandy Services
027         */
028        public String getKandyHostURL();
029
030        /**
031         * Get full configuration information about Kandy SDK, Kandy API, Kandy User Info and SPiDR SDK
032         * @return JSON's String representation with full configuration
033         */
034        public String getReport();
035        
036}
