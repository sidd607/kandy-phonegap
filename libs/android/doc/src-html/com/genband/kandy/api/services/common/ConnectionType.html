001package com.genband.kandy.api.services.common;
002
003public enum ConnectionType
004{
005        /**
006         * The absence of a connection type.
007         * 
008         */
009        NONE(-1),
010        /**
011         * The Mobile data connection. When active, all data traffic will use this
012         * network type's interface by default (it has a default route)
013         */
014        MOBILE(0),
015        /**
016         * The WIFI data connection. When active, all data traffic will use this
017         * network type's interface by default (it has a default route).
018         */
019        WIFI(1),
020        /**
021         * An MMS-specific Mobile data connection. This network type may use the
022         * same network interface as {@link #MOBILE} or it may use a different one.
023         * This is used by applications needing to talk to the carrier's Multimedia
024         * Messaging Service servers.
025         */
026//      MOBILE_MMS(2),
027        /**
028         * A SUPL-specific Mobile data connection. This network type may use the
029         * same network interface as {@link #MOBILE} or it may use a different one.
030         * This is used by applications needing to talk to the carrier's Secure User
031         * Plane Location servers for help locating the device.
032         */
033//      MOBILE_SUPL(3),
034        /**
035         * A DUN-specific Mobile data connection. This network type may use the same
036         * network interface as {@link #MOBILE} or it may use a different one. This
037         * is sometimes by the system when setting up an upstream connection for
038         * tethering so that the carrier is aware of DUN traffic.
039         */
040//      MOBILE_DUN(4),
041        /**
042         * A High Priority Mobile data connection. This network type uses the same
043         * network interface as {@link #MOBILE} but the routing setup is different.
044         * Only requesting processes will have access to the Mobile DNS servers and
045         * only IP's explicitly requested via {@link #requestRouteToHost} will route
046         * over this interface if no default route exists.
047         */
048//      MOBILE_HIPRI(5),
049        /**
050         * The WiMAX data connection. When active, all data traffic will use this
051         * network type's interface by default (it has a default route).
052         */
053//      WIMAX(6),
054        /**
055         * The Bluetooth data connection. When active, all data traffic will use
056         * this network type's interface by default (it has a default route).
057         */
058//      BLUETOOTH(7),
059        /**
060         * Dummy data connection. This should not be used on shipping devices.
061         */
062//      DUMMY(8),
063        /**
064         * The Ethernet data connection. When active, all data traffic will use this
065         * network type's interface by default (it has a default route).
066         */
067//      ETHERNET(9),
068        /**
069         * Over the air Administration.
070         */
071//      MOBILE_FOTA(10),
072        /**
073         * IP Multimedia Subsystem.
074         */
075//      MOBILE_IMS(11),
076        /**
077         * Carrier Branded Services.
078         */
079//      MOBILE_CBS(12),
080        /**
081         * A Wi-Fi p2p connection. Only requesting processes will have access to the
082         * peers connected.
083         */
084//      WIFI_P2P(13),
085        /**
086         * The network to use for initially attaching to the network
087         */
088//      MOBILE_IA(14),
089        /**
090         * All network type allowed
091         */
092        ALL(15);
093
094        private int value;
095
096        private ConnectionType(int value)
097        {
098                this.value = value;
099        }
100
101        public int getValue()
102        {
103                return this.value;
104        }
105
106}
