004import com.genband.kandy.communication.IKandyRestAPI;
005/**
006 * This interface is used to handle location info. (country codes)
007 */
008
009public interface IKandyLocationService extends IKandyRestAPI
010{
011
012        /**
013         * Get country code by MCC or client IP
014         * 
015         * @param listener The callback that will be invoked
016         */
017        public void getCountryInfo(KandyCountryInfoResponseListener listener);
018        
019        
020        /**
021         * Get single current location update automatically match the necessary provider.
022         * @param locationListener 
023         * @throws KandyIllegalArgumentException  in case and argument <tt>locationListener</tt> is null
024         */
025        public void getCurrentLocation(KandyCurrentLocationListener locationListener) throws KandyIllegalArgumentException;

027}
